Writing an Extension

Extensions are single .py files hosted on any URL. A ROM source extension must export EXTENSION_INFO and SOURCE_CLASS; a cover source exports COVER_SOURCE_CLASS instead. Host an index.json alongside the files to make them browseable here. 

EXTENSION_INFO = {
    "id": "my_source",
    "name": "My Source",
    "version": "1.0.0",
    "type": "rom_source",   # or "cover_source"
    "author": "you",
    "description": "A short description.",
}

from app.services.sources.base import RomSource

class MySource(RomSource):
    source_id = "my_source"
    name = "My Source"
    # ... implement search(), get_files(), get_download_url()

SOURCE_CLASS = MySource
